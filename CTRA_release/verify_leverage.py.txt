#!/usr/bin/env python3
"""Fourth Wall leverage verification — 18.33x"""

import json, os

class FourthWall:
    @staticmethod
    def send(fp, data):
        with open(fp, "w") as f:
            json.dump(data, f)
    @staticmethod
    def read(fp):
        try:
            with open(fp) as f:
                return json.load(f)
        except:
            return {}

def main():
    sf = "/tmp/ctra_leverage_demo.json"
    FourthWall.send(sf, {
        "internal_ticks": 0,
        "external_ops": 0,
        "observer_location": "[REDACTED]"
    })

    ext_ops = 3
    total_ticks = 55
    per = total_ticks // ext_ops

    print("=" * 60)
    print("  Fourth Wall Leverage Verification")
    print("=" * 60)

    for i in range(1, ext_ops + 1):
        s = FourthWall.read(sf)
        s["external_ops"] += 1
        FourthWall.send(sf, s)
        ticks = per + (1 if i == ext_ops and ext_ops * per < total_ticks else 0)
        s["internal_ticks"] += ticks
        FourthWall.send(sf, s)
        print(f"  Injection {i}: observer ops -> internal {ticks} steps")

    final = FourthWall.read(sf)
    ratio = final["internal_ticks"] / final["external_ops"]

    print("-" * 60)
    print(f"  Observer operations: {final['external_ops']}")
    print(f"  Internal inference steps: {final['internal_ticks']}")
    print(f"  Leverage ratio: {ratio:.2f}x")
    print(f"  Observer location: {final['observer_location']}")
    print("=" * 60)

    os.remove(sf)

if __name__ == "__main__":
    main()
